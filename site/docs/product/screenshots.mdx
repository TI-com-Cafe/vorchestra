export function ScreenshotGallery() {
  const [active, setActive] = useState(null);

  useEffect(() => {
    if (!active) return;
    const onKeyDown = (event) => {
      if (event.key === 'Escape') setActive(null);
    };
    document.addEventListener('keydown', onKeyDown);
    document.body.classList.add('lightboxOpen');
    return () => {
      document.removeEventListener('keydown', onKeyDown);
      document.body.classList.remove('lightboxOpen');
    };
  }, [active]);

  return (
    <>
      <div className="screenshotGallery">
        {screenshots.map((shot, index) => (
          <figure key={shot.src}>
            <button
              type="button"
              className="screenshotButton"
              onClick={() => setActive({...shot, index})}
              aria-label={`Maximize screenshot ${index + 1}: ${shot.alt}`}
            >
              <img src={shot.src} alt={shot.alt} loading="lazy" />
              <span className="screenshotZoomHint">Click to maximize</span>
            </button>
            <figcaption>{shot.caption}</figcaption>
          </figure>
        ))}
      </div>

      {active && (
        <div
          className="screenshotLightbox"
          role="dialog"
          aria-modal="true"
          aria-label={active.alt}
          onClick={() => setActive(null)}
        >
          <div className="screenshotLightboxPanel" onClick={(event) => event.stopPropagation()}>
            <div className="screenshotLightboxHeader">
              <div>
                <span>Screenshot {active.index + 1} of {screenshots.length}</span>
                <strong>{active.alt}</strong>
              </div>
              <button type="button" onClick={() => setActive(null)} aria-label="Close maximized screenshot">
                Close
              </button>
            </div>
            <img src={active.src} alt={active.alt} />
            <p>{active.caption}</p>
          </div>
        </div>
      )}
    </>
  );
}

<ScreenshotGallery />
